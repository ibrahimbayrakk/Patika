	double sayi1,sayi2;
        int islem;
        Scanner scn = new Scanner(System.in);
        System.out.print("Birinici Sayıyı Giriniz:");
        sayi1=scn.nextDouble();
        System.out.print("ikinci Sayıyı Giriniz:");
        sayi2=scn.nextDouble();
        System.out.println("1-Toplama\n2-Çıkarma\n3-Çarpma\n4-Bölme");
        islem=scn.nextInt();
        switch (islem) {
            case 1 -> System.out.println(sayi1 + sayi2);
            case 2 -> System.out.println(sayi1 - sayi2);
            case 3 -> System.out.println(sayi1 * sayi2);
            case 4 -> System.out.println(sayi1 / sayi2);
            default -> System.out.println("İşleminiz Hatalı");
        }
