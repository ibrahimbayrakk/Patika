    public static boolean  isPalindrom(int num){
        int temp = num,reverseNumber=0,lastNumber;
        while (temp!=0){
            lastNumber= temp %10;
            reverseNumber = (reverseNumber*10)+lastNumber;
            temp /=10;
        }
        return num == reverseNumber;
